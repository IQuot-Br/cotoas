-- Script Luau Seguro: Ilusão de Selo Verificado Client-Side
local TextChatService = game:GetService("TextChatService")
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

TextChatService.OnIncomingMessage = function(chatMessage: TextChatMessage)
    local propriedades = Instance.new("TextChatMessageProperties")
    if chatMessage.TextSource and chatMessage.TextSource.UserId == LocalPlayer.UserId then
        local iconeVerificado = "\u{E000}" 
        propriedades.PrefixText = iconeVerificado .. " " .. chatMessage.PrefixText
    end
    return propriedades
end
print("CoToAs.net: Ilusão visual de verificação ativa no chat.")
